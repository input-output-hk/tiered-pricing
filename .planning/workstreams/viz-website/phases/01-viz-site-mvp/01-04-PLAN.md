def parse_args() -> argparse.Namespace:
    # Existing: --source, --output, --include, --exclude
    # Added by this plan:
    #   --serve : action="store_true", default=False
    #   --port  : int, default=8000
    pass

def copy_static_assets(static_src: Path, output: Path) -> None:
    """Copy `static_src` (= `sim-rs/scripts/viz/static/`) into <output>/static/,
    and copy <static_src>/index.html into <output>/index.html.
    Idempotent — uses shutil.copy2 to overwrite older bundles.
    """

def serve(output_dir: Path, port: int) -> None:
    """Block-serve `output_dir` on 127.0.0.1:<port>.

    Uses http.server.ThreadingHTTPServer + functools.partial(SimpleHTTPRequestHandler, directory=str(output_dir)).
    Catches KeyboardInterrupt for clean Ctrl-C exit. Sets allow_reuse_address = True so re-runs don't fail with
    "Address already in use" on quick restarts.
    """

def main() -> None:
    args = parse_args()
    warnings = []
    run_build(args.source, args.output, args.include, args.exclude, warnings)
    copy_static_assets(Path(__file__).resolve().parent / "static", args.output)
    # Print warnings summary
    if args.serve:
        serve(args.output, args.port)

| Boundary | Description |
|----------|-------------|
| local HTTP socket → LAN | Mis-bound address would expose `sim-rs/output/` (which may contain not-yet-shared experiment results) to other machines on the developer's LAN. |
| developer-supplied `--output` path → filesystem | Argparse passes `--output` to file-IO; pathological paths (`../../../etc/`) could attempt traversal, but the dev owns the machine. |

| T-01-04-03 | Tampering | `--output` path traversal | accept | The user runs the script on their own filesystem. Path traversal into system directories is a user-error, not an attack; v1 does not lock down `Path(args.output).resolve()`. (RESEARCH.md Security Domain assigns this LOW priority.) |
